CREATE DATABASE IF NOT EXISTS educadata;
USE educadata;

-- Tabela Registros da base de dados
CREATE TABLE IF NOT EXISTS registro (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ano INT NOT NULL,
    codigo_uf_escola INT NOT NULL,
    sigla_uf_escola CHAR(2) NOT NULL,
    codigo_municipio_escola INT NOT NULL,
    nome_municipio_escola VARCHAR(100) NOT NULL,
    codigo_escola_educacenso BIGINT NOT NULL,
    nome_escola_educacenso VARCHAR(200) NOT NULL,
    dependencia_adm INT NOT NULL,
    localizacao_escola INT NOT NULL,
    matriculas INT NOT NULL,
    participantes_nec_esp INT NULL,
    participantes INT NOT NULL,
    taxa_participacao DOUBLE NULL,
    media_cn DOUBLE NULL,
    media_ch DOUBLE NULL,
    media_lp DOUBLE NULL,
    media_mt DOUBLE NULL,
    media_red DOUBLE NULL,
    media_obj DOUBLE NULL,
    media_tot DOUBLE NULL,
    inse DOUBLE NULL,
    pc_formacao_docente DOUBLE NULL,
    taxa_permanencia DOUBLE NULL,
    taxa_aprovacao DOUBLE NULL,
    taxa_reprovacao DOUBLE NULL,
    taxa_abandono DOUBLE NULL,
    porte_escola VARCHAR(50) NULL
);
